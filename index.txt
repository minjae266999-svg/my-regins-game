    function adjustTempStat(key, amount) {
        if (amount > 0 && statAdjustmentPoints > 0 && tempStats[key] < 100) { tempStats[key]++; statAdjustmentPoints--; }
        else if (amount < 0 && tempStats[key] > 0) { tempStats[key]--; statAdjustmentPoints++; }
        renderStatControls();
    }
